1) Variable-Width Column Chart

Height = mean flipper length (mm); Width = sample share per species.

vw <- peng |>
  group_by(species) |>
  summarise(n = n(),
            mean_flipper = mean(flipper_length_mm, na.rm = TRUE),
            .groups = "drop") |>
  arrange(desc(n)) |>
  mutate(w_share = n / sum(n),
         xmax = cumsum(w_share),
         xmin = lag(xmax, default = 0),
         xmid = (xmin + xmax)/2,
         fill = c(col_setosa, col_versicolor, col_virginica))

p1 <- ggplot(vw) +
  geom_rect(aes(xmin = xmin, xmax = xmax, ymin = 0, ymax = mean_flipper),
            fill = col_fill_soft, color = "#8C9A96", linewidth = 0.6) +
  geom_text(aes(x = xmid, y = mean_flipper + 5, label = species),
            size = 4, color = "#333333") +
  scale_y_continuous("Mean flipper length (mm)",
                     expand = expansion(mult = c(0, .12))) +
  scale_x_continuous("Share of samples",
                     labels = label_percent(accuracy = 1),
                     breaks = seq(0,1,0.2),
                     expand = expansion(mult = c(0,0))) +
  labs(title = "Average Flipper Length with Population-Weighted Widths",
       subtitle = "Each column's width is proportional to species' sample share",
       caption = "Data: palmerpenguins") +
  coord_cartesian(clip = "off")

2) Table with Embedded Charts

Small-multiple histograms (Species × Measurement). Muted fills; units shown in facet labels.

peng_long <- peng |>
  select(species, bill_length_mm, bill_depth_mm, flipper_length_mm, body_mass_g) |>
  pivot_longer(-species, names_to = "measurement", values_to = "value") |>
  mutate(measurement = factor(measurement,
         levels = c("bill_length_mm","bill_depth_mm","sepal_dummy","flipper_length_mm","body_mass_g"),
         labels = c("Bill length (mm)","Bill depth (mm)","_","Flipper length (mm)","Body mass (g)")))

# Use species-specific muted colors
fill_map <- c("Adelie" = col_setosa, "Chinstrap" = col_versicolor, "Gentoo" = col_virginica)

p2 <- ggplot(peng_long, aes(value, fill = species)) +
  geom_histogram(color = "white", bins = 12) +
  facet_grid(species ~ measurement, scales = "free_x") +
  scale_fill_manual(values = fill_map, guide = "none") +
  labs(title = "Distributions of Penguin Measurements",
       x = "Measurement Value", y = "Count",
       subtitle = "Each row is a species; each column is a measurement",
       caption = "Data: palmerpenguins") +
  theme(
    strip.text = element_text(face = "bold"),
    panel.grid = element_blank(),
    panel.border = element_rect(color = "#E6E6E6", fill = NA)
  )

3) Bar Chart

Two parallel charts (same design) so it’s easier to read — Adelie and Chinstrap body mass for every observed penguin.

many <- peng |>
  dplyr::filter(species %in% c("Adelie","Chinstrap"), !is.na(body_mass_g)) |>
  dplyr::group_by(species) |>
  dplyr::arrange(body_mass_g, .by_group = TRUE) |>
  dplyr::mutate(id = factor(row_number())) |>
  dplyr::ungroup()

fill_map <- c("Adelie" = col_setosa, "Chinstrap" = col_versicolor)

p3 <- ggplot(many, aes(id, body_mass_g, fill = species)) +
  geom_col(width = 0.65, color = "white") +
  coord_flip() +
  facet_wrap(~ species, ncol = 2, scales = "free_y") +
  scale_fill_manual(values = fill_map, guide = "none") +
  scale_y_continuous("Body mass (g)",
                     breaks = scales::pretty_breaks(n = 6),
                     labels = label_number(accuracy = 1, scale_cut = cut_si("")),
                     expand = expansion(mult = c(0.02, 0.06))) +
  labs(title = "Body Mass by Individual — Many Items, Few Categories",
       subtitle = "Each panel shows one species; bars ordered by body mass",
       x = NULL, caption = "Data: palmerpenguins") +
  theme(
    axis.text.y = element_blank(), axis.ticks.y = element_blank(),
    axis.text.x = element_text(color = "#333333")
  )

4) Column Chart

Few items shown cleanly: means of four measurements by species, faceted to respect units.

means_long <- peng |>
  dplyr::group_by(species) |>
  dplyr::summarise(
    `Bill depth (mm)`     = mean(bill_depth_mm,     na.rm = TRUE),
    `Bill length (mm)`    = mean(bill_length_mm,    na.rm = TRUE),
    `Body mass (g)`       = mean(body_mass_g,       na.rm = TRUE),
    `Flipper length (mm)` = mean(flipper_length_mm, na.rm = TRUE),
    .groups = "drop"
  ) |>
  tidyr::pivot_longer(-species, names_to = "measurement", values_to = "mean_value") |>
  dplyr::group_by(measurement) |>
  dplyr::mutate(index = 100 * mean_value / mean(mean_value, na.rm = TRUE)) |>
  dplyr::ungroup()

# 2) Manual x positions -> bars touch inside each trio, gap between trios
meas_levels <- c("Bill depth (mm)", "Bill length (mm)", "Body mass (g)", "Flipper length (mm)")
spec_levels <- c("Adelie","Chinstrap","Gentoo")

bars_per_group <- length(spec_levels)  # 3
gap_units      <- 1                    # tweak this to widen/narrow gaps between categories

plot_df <- means_long |>
  dplyr::mutate(
    measurement = factor(measurement, levels = meas_levels),
    species     = factor(species,     levels = spec_levels),
    g_idx = as.integer(measurement) - 1,
    s_idx = as.integer(species)     - 1,
    x     = g_idx * (bars_per_group + gap_units) + s_idx
  )

# axis breaks at the center of each trio
breaks <- ((0:(length(meas_levels)-1)) * (bars_per_group + gap_units)) + (bars_per_group - 1)/2
labels <- meas_levels

# 3) Muted palette
pal_grouped <- c("Adelie"="#5B7EA3", "Chinstrap"="#B58B45", "Gentoo"="#6BAA75")

p4 <- ggplot(plot_df, aes(x = x, y = index, fill = species)) +
  geom_col(width = 1, color = NA) +  # 0.98 so bars in a trio visually touch
  scale_x_continuous(breaks = breaks, labels = labels, expand = expansion(mult = c(0.02, 0.04))) +
  scale_fill_manual(values = pal_grouped, name = "Species") +
  scale_y_continuous("Index (measurement mean = 100)", expand = expansion(mult = c(0, 0.05))) +
  labs(
    title = "Mean Penguin Measurements by Species (Indexed)",
    x = "Measurement",
    caption = "Data: palmerpenguins"
  ) +
  theme_minimal(base_size = 13) +
  theme(
    plot.title  = element_text(face = "bold", size = 18),
    axis.text.x = element_text(angle = 12, hjust = 1, vjust = 1),
    panel.grid.minor = element_blank()
  )
