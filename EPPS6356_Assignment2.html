2 Exercise using the Happy Planet Index (HPI)

2.1 Base R Functions Used

  1. par() – Sets or queries graphical parameters such as margins, axis style, and layout.
  2. lines() – Adds connected line segments to an existing plot.
  3. points() – Adds individual points (symbols) to an existing plot.
  4. axis() – Draws tick marks and axis labels on plots.
  5. box() – Adds a box (border) around the current plotting region.
  6. text() – Writes text inside the plotting area at specified coordinates.
  7. mtext() – Adds text in the margins (outer areas) of the plot.
  8. hist() – Creates histograms for frequency distributions.
  9. boxplot() – Displays data distributions by quartiles.
  10. legend() – Adds legends identifying plotted elements.
  11. persp() – Produces 3-D perspective surface plots.
  12. names() – Retrieves or sets the names of variables in a data frame.
  13. pie() – Creates pie-chart visualizations.

2.2 Plots and Results

Make sure the file HPI_data.csv is in the same folder as this .qmd.

# Read and normalize column names from the HPI CSV. This maps your Excel-like
# headers to the short names used below, and ensures numeric types.
HPI_raw <- read.csv("HPI_data.csv", stringsAsFactors = FALSE, check.names = TRUE)

pick <- function(df, ...) {
  opts <- c(...)
  hit <- opts[opts %in% names(df)]
  if (!length(hit)) stop("Missing expected columns: ", paste(opts, collapse=" | "))
  df[[hit[1]]]
}

HPI <- data.frame(
  GDP_capita = suppressWarnings(as.numeric(pick(HPI_raw, "GDP_capita", "GDP.per.capita...."))),
  LifeExp    = suppressWarnings(as.numeric(pick(HPI_raw, "LifeExp", "Life.Expectancy..years."))),
  Wellbeing  = suppressWarnings(as.numeric(pick(HPI_raw, "Wellbeing", "Ladder.of.life..Wellbeing...0.10."))),
  Carbon     = suppressWarnings(as.numeric(pick(HPI_raw, "Carbon", "Carbon.Footprint..tCO2e."))),
  HPI        = suppressWarnings(as.numeric(pick(HPI_raw, "HPI"))),
  Population = suppressWarnings(as.numeric(pick(HPI_raw, "Population", "Population..thousands."))),
  Continent  = pick(HPI_raw, "Continent")
)
# Scatter with regression line — small, solid points

# persp(): demo surface (with clear axes & ticks)
x <- seq(-10, 10, length = 30); y <- x
f <- function(x,y) {
  r <- sqrt(x^2 + y^2)
  out <- 10 * sin(r) / r
  out[!is.finite(out)] <- 10
  out
}
z <- outer(x, y, f)
par(mar = c(2, 2.5, 3, 2))
persp(x, y, z, theta = 30, phi = 30, expand = 0.5,
      ticktype = "detailed",
      xlab = "X", ylab = "Y", zlab = "Z",
      col = "lightblue", border = "gray30",
      main = "persp(): demo surface")
# OPTIONAL: Interpolated HPI surface using akima (only if enough data)
if (!requireNamespace("akima", quietly = TRUE)) {
  install.packages("akima")
}
if (requireNamespace("akima", quietly = TRUE)) {
  library(akima)
  df <- subset(HPI, is.finite(GDP_capita) & is.finite(Carbon) & is.finite(HPI))
  if (nrow(df) > 10 && length(unique(df$GDP_capita)) > 3 && length(unique(df$Carbon)) > 3) {
    interp_result <- with(df, akima::interp(x = GDP_capita, y = Carbon, z = HPI, duplicate = "mean"))
    par(mar = c(2, 2.5, 3, 2))
    persp(interp_result$x, interp_result$y, interp_result$z,
          theta = 30, phi = 30, expand = 0.5,
          ticktype = "detailed",
          xlab = "GDP per Capita", ylab = "Carbon (tCO2e/capita)", zlab = "HPI",
          col = "lightblue", border = "gray30",
          main = "HPI Surface: GDP vs Carbon (interpolated)")
  } else {
    message("Not enough distinct points for interpolation; skipping HPI surface.")
  }
}
